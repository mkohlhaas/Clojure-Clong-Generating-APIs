pacman -Ql zlib

zlib /usr/lib/
zlib /usr/lib/libz.a
zlib /usr/lib/libz.so
zlib /usr/lib/libz.so.1
zlib /usr/lib/libz.so.1.3.1

zlib /usr/share/
zlib /usr/share/licenses/

zlib /usr/share/licenses/zlib/LICENSE
zlib /usr/share/man/
zlib /usr/share/man/man3/
zlib /usr/share/man/man3/zlib.3.gz
